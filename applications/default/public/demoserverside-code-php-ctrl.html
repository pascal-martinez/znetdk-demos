<?php namespace app\controller; class DemoForm extends \AppController { static protected function action_validate() { // JSON Response $response = new \Response(); // Data validation $validator = new \app\validator\MyValidator(); if ($validator->validate()) { $response->setSuccessMessage('Server-side validation', 'The form data has been validated successfully.'); } else { $response->setFailedMessage('Server-side validation', $validator->getErrorMessage(), $validator->getErrorVariable()); } // Response returned to the front controller return $response; } }
